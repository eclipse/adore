95  void run()
96  {
97  if (writer_->canWriteMore())
98  {
99  adore::env::BorderBased::CombinedLaneGeometry clg;
100 
101  adore::env::BorderBased::BorderID bestMatchFromLastLFG;
102  bool bestMatchFromLastLFGValid = false;
103  bool reset_lane_matching = false;
104  if (reset_reader_->hasUpdate())
105  {
106  reset_reader_->getData(reset_lane_matching);
107  }
108  if(activate_lfg_rematching_ && lfg_.isValid() && !reset_lane_matching)//info from last time
109  {
110  //try to match ego state to old borders: do this before roadmap_.update()!
111  roadmap_.updateEgoState();
112  adore::env::BorderBased::Border* border =
113  lfg_.getBestMatchingBorder(roadmap_.getVehicleState()->getX(),
114  roadmap_.getVehicleState()->getY(),
115  2.0 * apLateralPlanner_->getHardSafetyDistanceToLaneBoundary()
116  + 0.5 * apVehicle_->get_bodyWidth());
117  if(border!=nullptr)
118  {
119  bestMatchFromLastLFG = border->m_id;
120  bestMatchFromLastLFGValid = true;
121  }
122  }
123 
124  roadmap_.update(bestMatchFromLastLFGValid,bestMatchFromLastLFG);
125  auto current_border = roadmap_.getMatchedBorder();
126  auto ego_state = roadmap_.getEgoState();
127 
128 
129 
130  lfg_.update(roadmap_.getBorderSet(), roadmap_.getBorderTrace(), roadmap_.getBorderCostMap(),
131  current_border, roadmap_.getVehicleState(), apLFV_->getLookAhead(),
132  apLFV_->getLookBehind(), apLFV_->getBaselineFitSmoothness(),
133  roadmap_.isNavigationActive());
134 
135  speedlimit_manager_.update(roadmap_.getVehicleState()->getX(),roadmap_.getVehicleState()->getY());
136  indicatorhint_manager_.update(roadmap_.getVehicleState()->getX(),roadmap_.getVehicleState()->getY());
137 
138  if(lfg_.isValid())
139  {
140  //copy to data object
141  // lfg_.m_centerSmoothed_fct
142  copy_lfg_to_lfg_proxy(clg.center, lfg_);
143 
144  speedlimit_manager_.getFunction(
145  clg.center->centerSmoothed_fct,
146  clg.center->leftDistance_fct,
147  clg.center->rightDistance_fct,
148  clg.center->speedLimit_fct);
149 
150  indicatorhint_manager_.getFunctions(
151  clg.center->centerSmoothed_fct,
152  clg.center->leftDistance_fct,
153  clg.center->rightDistance_fct,
154  clg.center->left_indicator_hint_fct,
155  clg.center->right_indicator_hint_fct);
156 
157  auto borders_in_lfg = lfg_.getRightBorders()->getBorders();
158  auto current_border_iterator = borders_in_lfg->begin();
159  auto first_border_iterator = borders_in_lfg->begin();
160  auto last_border_iterator = std::next(borders_in_lfg->begin(),borders_in_lfg->size()-1);
161  for(;(*current_border_iterator)!=current_border
162  && current_border_iterator!=borders_in_lfg->end();
163  current_border_iterator++){}
164  double offset_on_current = 0.0;
165 
166  if(current_border->m_path!=nullptr)
167  {
168  double n;
169  offset_on_current = current_border->m_path->getClosestParameter(ego_state.getX(),ego_state.getY(),1,2,n);
170  }
171 
172  if(current_border_iterator!=borders_in_lfg->end())
173  {
174  ilcg_left_->setSmoothness(apLCV_->getBaselineFitSmoothness());
175  ilcg_left_->setLookAhead(apLCV_->getLookAhead());
176  ilcg_left_->setLookBehind(apLCV_->getLookBehind());
177  ilcg_left_->setMaximumNavCostIncrease(apLCV_->getMaximumNavCostLoss());
178  ilcg_right_->setSmoothness(apLCV_->getBaselineFitSmoothness());
179  ilcg_right_->setLookAhead(apLCV_->getLookAhead());
180  ilcg_right_->setLookBehind(apLCV_->getLookBehind());
181  ilcg_right_->setMaximumNavCostIncrease(apLCV_->getMaximumNavCostLoss());
182  ilcg_left_->update(offset_on_current,current_border_iterator,first_border_iterator,last_border_iterator);
183  ilcg_right_->update(offset_on_current,current_border_iterator,first_border_iterator,last_border_iterator);
184  if(ilcg_left_->isValid())
185  {
186  copy_ilcg(clg.left,clg.leftChange,ilcg_left_);
187  speedlimit_manager_.getFunction(
188  clg.left->centerSmoothed_fct,
189  clg.left->leftDistance_fct,
190  clg.left->rightDistance_fct,
191  clg.left->speedLimit_fct);
192 
193  indicatorhint_manager_.getFunctions(
194  clg.left->centerSmoothed_fct,
195  clg.left->leftDistance_fct,
196  clg.left->rightDistance_fct,
197  clg.left->left_indicator_hint_fct,
198  clg.left->right_indicator_hint_fct);
199  }
200  else
201  {
202  clg.left->isValid = false;
203  clg.leftChange->isValid = false;
204  }
205 
206  if(ilcg_right_->isValid())
207  {
208  copy_ilcg(clg.right,clg.rightChange,ilcg_right_);
209  speedlimit_manager_.getFunction(
210  clg.right->centerSmoothed_fct,
211  clg.right->leftDistance_fct,
212  clg.right->rightDistance_fct,
213  clg.right->speedLimit_fct);
214 
215  indicatorhint_manager_.getFunctions(
216  clg.right->centerSmoothed_fct,
217  clg.right->leftDistance_fct,
218  clg.right->rightDistance_fct,
219  clg.right->left_indicator_hint_fct,
220  clg.right->right_indicator_hint_fct);
221  }
222  else
223  {
224  clg.right->isValid = false;
225  clg.rightChange->isValid = false;
226  }
227  }
228 
229  writer_->write(clg);
230  }
231  }

234  void copy_ilcg(std::shared_ptr<adore::env::BorderBased::LaneGeometryDataProxy> target_geometry,
235  std::shared_ptr<adore::env::BorderBased::LaneChangeDataProxy> target_lcdata,
236  adore::env::BorderBased::IndependentLaneChangeGeometry* lane )
237  {
238  target_geometry->centerSmoothed_fct = lane->baseline_.position_fct_;
239  target_geometry->centerNormal_fct = lane->baseline_.normal_fct_;
240  target_geometry->centerSmoothedCurvature_fct = lane->baseline_.curvature_fct_;
241  target_geometry->centerSmoothedCurvatureDerivative_fct = lane->baseline_.curvatureDerivative_fct_;
242  target_geometry->navigationCost_fct = lane->navigationCost_fct_;
243  target_geometry->isValid = true;
244  target_lcdata->gate_s0 = lane->getProgressGateOpen();
245  target_lcdata->gate_s1 = lane->getProgressGateClosed();
246  target_lcdata->direction = lane->lcb_.isLCDirectionLeft()
247  ? adore::view::ALaneChangeView::LEFT
248  : adore::view::ALaneChangeView::RIGHT;
249  target_lcdata->isValid = true;
250  target_lcdata->targetOuterBorderDistance_fct = lane->offsetTargetOuterBorders_;
251  target_lcdata->separatingBorderDistance_fct = lane->offsetSeparatingBorders_;
252  target_lcdata->sourceOuterBorderDistance_fct = lane->offsetSourceOuterBorders_;
253  if(lane->lcb_.isLCDirectionLeft())
254  {
255  target_geometry->leftDistance_fct = lane->offsetTargetOuterBorders_;
256  target_geometry->rightDistance_fct = lane->offsetSeparatingBorders_;
257  }
258  else
259  {
260  target_geometry->leftDistance_fct = lane->offsetSeparatingBorders_;
261  target_geometry->rightDistance_fct = lane->offsetTargetOuterBorders_;
262  }
263  }
