53  roadmap_(EnvFactory::get(),ParamsFactory::get()),three_lanes_(false/*do not monitor traffic*/),
54  connectionSet_(EnvFactory::get()->getControlledConnectionFeed()),
55  worstcase_(worstcase)

57  if(worstcase)
58  {
59  prediction_reader = EnvFactory::get()->getWorstCaseRawPredictionSetReader();
60  prediction_writer = EnvFactory::get()->getWorstCasePredictionSetWriter();
61  }
62  else
63  {
64  prediction_reader = EnvFactory::get()->getExpectedRawPredictionSetReader();

75  void run()
76  {
77  roadmap_.update();
78  three_lanes_.update();
79  if(motion_state_reader_->hasData())
80  {
81  adore::env::VehicleMotionState9d motion_state;
82  motion_state_reader_->getData(motion_state);
83  connectionSet_.update(pLocalRoadMap_->getDiscardRadius(),motion_state.getX(),motion_state.getY());
84 
85  if(prediction_reader->hasUpdate())
86  {
87  adore::env::OccupancyCylinderPredictionSet predictionInputSet;//the input set
88  prediction_reader->getData(predictionInputSet);//read the input set
89  adore::mad::OccupancyCylinderTree prediction_blocker;//positions, which block predictions are accumulated here
90 
91  adore::env::PrecedenceFilter precedenceFilter;
92  adore::env::PrecedenceFilter::TFilterResult relevant_rules;
93  //get all priority rules, which assign higher priority to left, right or center lane
94  //@TODO: decide whether to use separate prediction topics filtered for left/right/current lane planning
95  precedenceFilter.getHighPrioritiesOnLane(roadmap_.getPrecedenceSet(),three_lanes_.getCurrentLane(),relevant_rules);
96  precedenceFilter.getHighPrioritiesOnLane(roadmap_.getPrecedenceSet(),three_lanes_.getLeftLaneChange()->getTargetLane(),relevant_rules);
97  precedenceFilter.getHighPrioritiesOnLane(roadmap_.getPrecedenceSet(),three_lanes_.getRightLaneChange()->getTargetLane(),relevant_rules);
98  if(!worstcase_||pPrediction_->get_worstcase_filter_precedence())
99  {
100  for(auto rule:relevant_rules)
101  {
102  //place a blocker at low-priority connection start
103  adore::mad::OccupancyCylinder occupancy(0.5,rule->low_.from_(0),rule->low_.from_(1),-1.0e99,1.0e99,rule->low_.from_(2),rule->low_.from_(2)+3.0);
104  prediction_blocker.insert(occupancy);
105  }
106  }
107  //get controlled connections and block predictions which pass through connection-start-points associated with red-lights
108  if(!worstcase_||pPrediction_->get_worstcase_filter_tcd())
109  {
110  for(auto itpair = connectionSet_.getAllConnections();
111  itpair.current()!=itpair.end();
112  itpair.current()++)
113  {
114  adore::env::ControlledConnection* con = itpair.current()->second;
115  auto from = con->getID().getFrom();
116  //walk through all known phases
117  double t0 = motion_state.getTime()-5.0;
118  for(adore::env::ConnectionStateEvent& state:con->data_)
119  {
120  double t1 = state.getMinEndTime();
121  double nextt0 = state.getMaxOrMinEndTime();
122  //if not green, create obstacle
123  if(t1>t0 && blocking_states_.find(state.getState()) != blocking_states_.end())
124  {
125  adore::mad::OccupancyCylinder occupancy(0.5,from.get<0>(),from.get<1>(),t0,t1,from.get<2>()-2.0,from.get<2>()+3.0);
126  prediction_blocker.insert(occupancy);
127  }
128  t0 = nextt0;
129  }
130  }
131  }
132  //protect ego against tail-gating predictions
133  if(three_lanes_.getCurrentLane()->isValid())
134  {
135  //compute point behind ego to serve as stopping point for potential "tail-gating" predictions
136  double s,n;
137  three_lanes_.getCurrentLane()->toRelativeCoordinates(motion_state.getX(),motion_state.getY(),s,n);
138  // //based on current lane, behind ego
139  // for(double ds=0.0;ds<5.0;ds+=1.0)
140  // {
141  // double X,Y,Z;
142  // three_lanes_.getCurrentLane()->toEucledianCoordinates(s-ds,n,X,Y,Z);
143  // adore::mad::OccupancyCylinder occupancy(pVehicle_->get_bodyWidth(),X,Y,-1.0e99,1.0e99,Z-2.0,Z+3.0);
144  // prediction_blocker.insert(occupancy);
145  // }
146  double Z = motion_state.getZ();
147  //rear end of ego
148  double X0 = motion_state.getX()-std::cos(motion_state.getPSI())*pVehicle_->get_d();
149  double Y0 = motion_state.getY()-std::sin(motion_state.getPSI())*pVehicle_->get_d();
150  //front end of ego
151  double X1 = motion_state.getX()+std::cos(motion_state.getPSI())*(pVehicle_->get_a()+pVehicle_->get_b()+pVehicle_->get_c());
152  double Y1 = motion_state.getY()+std::sin(motion_state.getPSI())*(pVehicle_->get_a()+pVehicle_->get_b()+pVehicle_->get_c());
153  double w = pVehicle_->get_bodyWidth()*0.5;
154  double r = w + 0.05;
155  double dr = std::sqrt(r*r-w*w);
156  double dx = X1-X0;
157  double dy = Y1-Y0;
158  double L = std::sqrt(dx*dx+dy*dy);
159  for(s = 0;s<=L*0.5;s+=2.0*dr)
160  {
161  adore::mad::OccupancyCylinder occupancy(r,X0+s/L*dx,Y0+s/L*dy,-1.0e99,1.0e99,-2.0,+3.0);//HeD, 20210521: TODO insert hight as soon as prediction provider also provides correct object hight
162  // adore::mad::OccupancyCylinder occupancy(r,X0+s/L*dx,Y0+s/L*dy,-1.0e99,1.0e99,Z-2.0,Z+3.0);
163  prediction_blocker.insert(occupancy);
164  }
165 
166  }
167 
168  //find all prediction branches of the input set, which are colliding with a blocker: clip at collision time
169  std::unordered_map<int,double> clipped_branches;
170  std::unordered_set<int> blocked_branches;
171  for(auto& prediction:predictionInputSet)
172  {
173  double clip_time;
174  if(prediction.occupancy_.getEarliestCollisionTime(prediction_blocker,1.0e99,clip_time))
175  {
176  clipped_branches.emplace(std::make_pair(prediction.branchID_,clip_time));
177  }
178  }
179 
180  //compute all successors of blocked or clipped branches
181  auto changed = true;
182  while(changed)
183  {
184  changed = false;
185  for(const auto& prediction:predictionInputSet)
186  {
187  if( ( clipped_branches.find(prediction.predecessorID_)!=clipped_branches.end()
188  || blocked_branches.find(prediction.predecessorID_)!=blocked_branches.end() )
189  && blocked_branches.find(prediction.branchID_)==blocked_branches.end())
190  {
191  changed = true;
192  blocked_branches.emplace(prediction.branchID_);
193  }
194  }
195  }
196 
197  //gather all predictions, which are neither blocked nor clipped
198  adore::env::OccupancyCylinderPredictionSet predictionOutputSet;//the filtered output set
199  for(const auto& prediction:predictionInputSet)
200  {
201  if( blocked_branches.find(prediction.branchID_)==blocked_branches.end()
202  && clipped_branches.find(prediction.branchID_)==clipped_branches.end() )
203  {
204  predictionOutputSet.push_back(prediction);
205  }
206  }
207  //gather clipped predictions
208  for(const auto& prediction:predictionInputSet)
209  {
210  auto branch_clipped = clipped_branches.find(prediction.branchID_);
211  if( branch_clipped!=clipped_branches.end()
212  && blocked_branches.find(prediction.branchID_)==blocked_branches.end() )
213  {
214  double clip_time = branch_clipped->second;
215  adore::env::OccupancyCylinderPrediction clipped_prediction;
216  clipped_prediction.branchID_ = prediction.branchID_;
217  clipped_prediction.trackingID_ = prediction.trackingID_;
218  clipped_prediction.predecessorID_ = prediction.predecessorID_;
219  clipped_prediction.confidence_ = prediction.confidence_;
220  for(const auto& indexed_cylinder:prediction.occupancy_.getLevel(0))
221  {
222  const auto& cylinder = indexed_cylinder.second;
223  if(cylinder.t0_<clip_time)
224  {
225  clipped_prediction.occupancy_.insert(cylinder);
226  }
227  }
228  if(clipped_prediction.occupancy_.getOccupancyCount()>0)
229  {
230  predictionOutputSet.push_back(clipped_prediction);
231  }
232  }
233  }
234 
235  //send gathered predictions
236  prediction_writer->write(predictionOutputSet);
237  }
238 
239 
240  }
241  }
242  };
243 
244  }
