32 class ConflictZone : public adore::view::ConflictZone
33 {
34 private:
35  std::vector<Border *> pathOfCrossTraffic_; // vector which contains the borders that lead to the conflict zone
36  std::vector<Coordinate> coordinates_; // vecotr of coordinates of the intersection points
37  double firstBorderDistToExit_; // distance from the end of first Border in pathOfCrossTraffic_ to the end of the conflict zone
38  double firstBorderDistToEntr_; // distance from the end of first Border in pathOfCrossTraffic_ to the beginning of the conflict zone
39  bool distancesCalculated_; // flag whether exitDistance_ and entryDistance_ are already calculated
40 
41 public:
42  ConflictZone()
43  {
44  distancesCalculated_ = false;
45  }
46  void setPathOfCrossTraffic(std::vector<Border *> poct)
47  {
48  pathOfCrossTraffic_.clear();
49  pathOfCrossTraffic_ = poct;
50  }
51  std::vector<Border *> *getPathOfCrossTraffic()
52  {
53  return &pathOfCrossTraffic_;
54  }
55  void setCoordinates(std::vector<Coordinate> c)
56  {
57  coordinates_ = c;
58  }
59  void calcDistances(BorderSet *bs)
60  {
61  // exit distance
62  firstBorderDistToExit_ = std::numeric_limits<double>::min();
63  double s, n;
64  std::map<double, Border *> possibleEntrances;
65  for (auto &c : coordinates_)
66  {
67  firstBorderDistToExit_ = std::max(firstBorderDistToExit_, (*pathOfCrossTraffic_.begin())->m_path->getClosestParameter(c.m_X, c.m_Y, 1, 2, n));
68  }
69  firstBorderDistToExit_ = (*pathOfCrossTraffic_.begin())->m_path->limitHi() - firstBorderDistToExit_;
70 
71  // entry distance
72  firstBorderDistToEntr_ = std::numeric_limits<double>::min();
73 
74  for (auto &c : coordinates_)
75  {
76  for (auto &b : pathOfCrossTraffic_)
77  {
78  b->m_path->getPositionOfPoint(c.m_X, c.m_Y, 1, 2, s, n);
79  if (s < 0 || s > 1)
80  {
81  continue;
82  }
83  possibleEntrances.insert(std::make_pair(b->m_path->getClosestParameter(c.m_X, c.m_Y, 1, 2, n), b));
84  break;
85  }
86  }
87  std::vector<double> dist;
88  double smax;
89  for (auto &p : possibleEntrances)
90  {
91  auto cpPath = pathOfCrossTraffic_;
92  dist.clear();
93  bs->getDistancesBetweenBordersAlongSuccessors(p.second, *cpPath.begin(), &cpPath, dist);
94  if (!dist.empty())
95  {
96  smax = *std::max_element(dist.begin(), dist.end());
97  }
98  else
99  {
100  smax = 0;
101  }
102  if (*pathOfCrossTraffic_.begin() != p.second)
103  {
104  smax += p.second->m_path->limitHi();
105  }
106  firstBorderDistToEntr_ = std::max(firstBorderDistToEntr_, (*pathOfCrossTraffic_.begin())->m_path->limitHi() + smax - p.first);
107  }
108  distancesCalculated_ = true;
109  }
110  double getFirstBorderExitDistance() // todo rename this
111  {
112  return distancesCalculated_ ? firstBorderDistToExit_ : 0.0;
113  }
114  double getFirstBorderEntranceDistance() // todo rename this
115  {
116  return distancesCalculated_ ? firstBorderDistToEntr_ : 0.0;
117  }
118 };

242  */

279  skip_this_path = true;
280  break;
281  }
282  }
283  if (skip_this_path)
284  continue;
285 
289  bool new_overlap_set = true; // form a new overlapset if the overlap can't be added to existing overlapset
290  for (auto &ols : overlap_sets)
291  {
292  if (std::find(ols.conflictingBorders_.begin(),
293  ols.conflictingBorders_.end(), overlap->m_second_right) !=
294  ols.conflictingBorders_.end())
295  {
296  // overlap is added to overlapset that contains the the first conflicting border of the overlap
297  ols.borderoverlaps_.push_back(&(*overlap));
298  new_overlap_set = false;
299  break;
300  }
301  if (std::find_first_of(path_of_conflicting_veh.begin(),
302  path_of_conflicting_veh.end(),
303  ols.conflictingBorders_.begin(),
304  ols.conflictingBorders_.end()) !=
305  path_of_conflicting_veh.end())
306  {
307  // overlap is added to overlapset that contains the any conflicting border of the overlap
308  bool update_conflicting_path = false;
309  ols.borderoverlaps_.push_back(&(*overlap));
310  for (auto cborder : path_of_conflicting_veh)
311  {
312  for (auto oborder : right_borders_of_conf_lanes)
313  {
314  if (cborder == oborder && std::find(ols.conflictingBorders_.begin(), ols.conflictingBorders_.end(), cborder) == ols.conflictingBorders_.end())
315  {
316  // update the conflicting path and conflicting borders of overlap set
317  ols.conflictingBorders_.push_back(cborder);
318  update_conflicting_path = true;
319  }
320  }
321  }
322  if(update_conflicting_path)
323  {
324  ols.conflictingPath_ = path_of_conflicting_veh;
325  }
326  new_overlap_set = false;
327  break;
328  }
329  }
330  if (new_overlap_set)
331  {
332 
333  BorderOverlapSet bos;
334  bos.borderoverlaps_.push_back(&(*overlap));
335  // add conjunction of boderoverlaps' path and
336  // right_borders_of_conf_lanes
337 
338  for (auto cborder : path_of_conflicting_veh)
339  {
340  for (auto oborder : right_borders_of_conf_lanes)
341  {
342  if (cborder == oborder)
343  bos.conflictingBorders_.push_back(cborder);
344  }
345  }
346  bos.conflictingPath_ = path_of_conflicting_veh;
347  overlap_sets.push_back(bos);
348  }
349  }

354  {
355  bool delete_this_ols = false;
356  bool left_left_intersection = false;
357  bool right_left_intersection = false;
358  bool left_right_intersection = false;
359  bool right_right_intersection = false;
360  for (auto &ol : ols->borderoverlaps_)
361  {

365  right_right_intersection = right_right_intersection || ol->m_has_intersection_right_right;
366  }
367  if (left_left_intersection + right_left_intersection + left_right_intersection + right_right_intersection == 1)
368  {
369  double n = 0;
370  double n_new = 0;
371  double s = 0;
372  if (left_left_intersection)
373  {
374  for (auto &ol : ols->borderoverlaps_)

378  n_new = lfg_->getOffsetOfLeftBorder(s) - n_new;

394  c = ol->m_second_right->m_path->f(0);

400  if (right_left_intersection)

402  for (auto &ol : ols->borderoverlaps_)
403  {

406  n_new = -lfg_->getOffsetOfRightBorder(s) + n_new;
407  n = std::max(n, n_new);
408  c = ol->m_second_left->m_path->f(0);
409  lfg_->toRelativeCoordinates(c(0), c(1), s, n_new);

413  }
414  if (right_right_intersection)
415  {
416  for (auto &ol : ols->borderoverlaps_)
417  {

423  lfg_->toRelativeCoordinates(c(0), c(1), s, n_new);
424  n_new = -lfg_->getOffsetOfRightBorder(s) + n_new;

441  }
442  }

444  assembleConflictZones(precedenceSet, borderSet, rightBorders, leftBorders);
445  mapParticipantsToConflictZones(elt, borderSet);
446  valid_ = true;

448  void assembleConflictZones(PrecedenceSet *precedenceSet, BorderSet *borderSet, BAContainer *rightBorders, BAContainer *leftBorders)
449  {
450  std::vector<Coordinate> coordinates;
451  std::multimap<double, ConflictZone *> conflictzones;
452  double s, n;
453  // look for segments of unary priority rules for ego vehicle
454  std::vector<std::pair<double, double>> egoHasPriority; // pairs in vector represent segments in s coordinates in which ego vehicle has priority over all other
455  auto rules = precedenceSet->getAllRules();
456  for (auto rule= rules->begin();rule!=rules->end();++rule)
457  {
458  auto pr = rule->second;
459  if (!pr->unary_)
460  continue;
461  auto highFrom = pr->high_.from_;
462  auto highFromSet = borderSet->getBordersAtPoint(highFrom(0), highFrom(1));
463  if (std::find_first_of(highFromSet.begin(), highFromSet.end(), rightBorders->begin(), rightBorders->end()) != highFromSet.end())
464  {
465  auto highTo = pr->high_.to_;
466  auto highToSet = borderSet->getBordersAtPoint(highTo(0), highTo(1));
467  if (std::find_first_of(highToSet.begin(), highToSet.end(), rightBorders->begin(), rightBorders->end()) != highToSet.end())
468  {
469  double s_from,s_to,n;
470  lfg_->toRelativeCoordinates(highFrom(0),highFrom(1),s_from,n);
471  lfg_->toRelativeCoordinates(highTo(0),highTo(1),s_to,n);
472  egoHasPriority.push_back(std::make_pair(s_from,s_to));
473  }
474  }
475  }
476 
477 
478  for (auto &os : overlap_sets)
479  {
480  coordinates.clear();
481  os.getCoordinates(coordinates);
482  // determine s_min, s_max
483  double s_min = std::numeric_limits<double>::max();
484  double s_max = std::numeric_limits<double>::min();
485  double x_min = std::numeric_limits<double>::max();
486  double x_max = std::numeric_limits<double>::min();
487  double y_min = std::numeric_limits<double>::max();
488  double y_max = std::numeric_limits<double>::min();
489  for (auto c : coordinates)
490  {
491  lfg_->toRelativeCoordinates(c.m_X, c.m_Y, s, n);
492  s_min = std::min(s_min, s);
493  s_max = std::max(s_max, s);
494  x_min = std::min(x_min, c.m_X);
495  x_max = std::max(x_max, c.m_X);
496  y_min = std::min(y_min, c.m_Y);
497  y_max = std::max(y_max, c.m_Y);
498  }
499  env::BorderBased::ConflictZone *cz = new env::BorderBased::ConflictZone();
500  cz->setStartProgress(s_min);
501  cz->setEndProgress(s_max);
502  cz->setWaitingPosition(s_min);
503  cz->setCoordinates(coordinates);
504  cz->setPathOfCrossTraffic(os.conflictingPath_);
505  // check whether cz has end
506 
507  double x_end = 0.5*((*os.conflictingPath_.begin())->m_id.m_last.m_X + (*os.conflictingPath_.begin())->m_left->m_last.m_X);
508  double y_end = 0.5*((*os.conflictingPath_.begin())->m_id.m_last.m_Y + (*os.conflictingPath_.begin())->m_left->m_last.m_Y);
509  auto end_border=borderSet->getBordersAtPoint(x_end,y_end);
510  if (std::find_first_of(end_border.begin(),end_border.end(),rightBorders->begin(),rightBorders->end())!=end_border.end())
511  {
512  cz->setHasEndProgress(false);
513  }
514  else
515  {
516  cz->setHasEndProgress(true);
517  }
518  for (auto &prioritysegment:egoHasPriority)
519  {
520  if (s_min > prioritysegment.first && s_max < prioritysegment.second)
521  {
522  cz->setPriority(false); // conflict zone is within a segment in which ego has right of way
523  break;
524  }
525  }
526  if (cz->hasPriority())
527  {
528  auto ruleitpair = precedenceSet->getRulesInRegion(x_min, y_min, x_max, y_max);
529  for (; ruleitpair.first != ruleitpair.second; ++ruleitpair.first)
530  {
531  auto pr = (*ruleitpair.first).second;
532  if (pr->unary_)
533  continue;
534  auto from = pr->low_.from_;
535  auto fromSet = borderSet->getBordersAtPoint(from(0), from(1));
536  if (std::find_first_of(fromSet.begin(),
537  fromSet.end(),
538  os.conflictingPath_.begin(),
539  os.conflictingPath_.end()) != fromSet.end())
540  {
541  std::cout << " found from point in conflicting path"<<std::endl;
542  // check whether ego lane is high route
543  auto highFrom = pr->high_.from_;
544  auto highFromSet = borderSet->getBordersAtPoint(highFrom(0), highFrom(1));
545  if (std::find_first_of(highFromSet.begin(), highFromSet.end(), rightBorders->begin(), rightBorders->end()) != highFromSet.end())
546  {
547  auto highTo = pr->high_.to_;
548  auto highToSet = borderSet->getBordersAtPoint(highTo(0), highTo(1));
549  if (std::find_first_of(highToSet.begin(), highToSet.end(), rightBorders->begin(), rightBorders->end()) != highToSet.end())
550  {
551  cz->setPriority(false);
552  break;
553  }
554  }
555  }
556  // pr->low_.to_; // TODO commented out to fix -Wunused-value, bring back in when needed
557  //TODO check to_-Point of priority route with successors of first border in cz->getPathOfCrossTraffic_
558  }
559  }
560  conflictzones.insert(std::make_pair(s_min, cz));
561  }
562  // add conflictzones to conflict area
563  for (auto cz = conflictzones.begin(); cz != conflictzones.end(); ++cz)
564  {
565  //if (!ca_.empty())
566  //std::cout << "cz.first="<<cz->first<< " caback="<<ca_.back()->getEndProgress()<<std::endl;
567  if (ca_.empty() || cz->first - ca_.back()->getEndProgress() > 5)
568  {
569  ConflictArea *ca = new ConflictArea();
570  ca->addConflictZone(cz->second);
571  ca_.push_back(ca);
572  continue;
573  }
574  ca_.back()->addConflictZone(cz->second);
575  }

577  void mapParticipantsToConflictZones(adore::env::traffic::EgoLaneTraffic * elt, BorderSet * borderSet)
578  {
579  // get the traffic participants
580  auto trafficMap = elt->getTrafficMap();
581  for (auto &ca : ca_)
582  {
583  auto czs = ca->getBorderBasedConflictZones();
584  for (adore::env::BorderBased::ConflictZone *&cz : *czs)
585  {
586  auto cp = cz->getPathOfCrossTraffic();
587  double length = 0.0;
588  for (auto &b : *cp)
589  {
590  length += b->m_path->limitHi();
591  }
592  cz->calcDistances(borderSet);
593  auto queue = cz->getCrossTraffic();
594  queue->clear();
595  std::unordered_set<env::traffic::Participant::TTrackingID> idset;
596  for (auto &b : *cp)
597  {
598  for (auto it = trafficMap->getBorderToParticipant().find(b->m_id);
599  it != trafficMap->getBorderToParticipant().end();
600  it++)
601  {
602  if (!it->second.first.m_isReferenceInside)
603  continue;
604  std::vector<double> distances;
605  auto cpcopy = *cp;
606  borderSet->getDistancesBetweenBordersAlongSuccessors(
607  borderSet->getBorder(it->first), *cpcopy.begin(), &cpcopy, distances);
608  double smin = 0;
609  if (!distances.empty())
610  {
611  smin = *std::min_element(distances.begin(), distances.end());
