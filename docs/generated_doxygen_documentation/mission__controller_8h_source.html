63  void run()
64  {
65  //update data
66  motion_state_reader_->getData(motion_state_);
67  x_state_reader_->getData(x_state_);
68  nav_goal_reader_->getData(nav_goal_);
69 
70  double distance_to_goal = std::sqrt((motion_state_.getX()-nav_goal_.target_.x_)*(motion_state_.getX()-nav_goal_.target_.x_)
71  + (motion_state_.getY()-nav_goal_.target_.y_)*(motion_state_.getY()-nav_goal_.target_.y_));
72 
73  //state machine
74  switch(mission_data_.getMissionState())
75  {
76  case adore::fun::MissionData::AUTOMATION_NOT_ACTIVE:
77  {
78  if(x_state_.getAutomaticControlAccelerationOn() && distance_to_goal > p_mission_control_->getDistanceToGoalForStart())
79  {
80  mission_data_.setMissionState(adore::fun::MissionData::DRIVING_TO_GOAL);
81  }
82  }
83  break;
84  case adore::fun::MissionData::DRIVING_TO_GOAL:
85  {
86  if(distance_to_goal < p_mission_control_->getDistanceToGoalForStop())
87  {
88  mission_data_.setMissionState(adore::fun::MissionData::GOAL_REACHED);
89  }
90  }
91  break;
92  case adore::fun::MissionData::GOAL_REACHED:
93  {
94  if(distance_to_goal > p_mission_control_->getDistanceToGoalForStart())
95  {
96  mission_data_.setMissionState(adore::fun::MissionData::DRIVING_TO_GOAL);
97  }
98  }
99  break;
100  }
101 
102  //reset mission state, if automation is not actived
103  if(!x_state_.getAutomaticControlAccelerationOn())
104  {
105  mission_data_.setMissionState(adore::fun::MissionData::AUTOMATION_NOT_ACTIVE);
106  }
107 
108  //write mission data
109  mission_data_writer_->write(mission_data_);
110  }
111  };
