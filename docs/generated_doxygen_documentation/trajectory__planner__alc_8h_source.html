196  virtual void computeTrajectory(const adore::fun::PlanningRequest& planning_request, adore::fun::PlanningResult& planning_result) override
197  {
198  //document planner result
199  planning_result.id = id_; // in lfbehavior, there is only one maneuver
200  planning_result.name = plannerName_;
201  planning_result.maneuver_type = adore::fun::PlanningResult::NOMINAL_DRIVING;
202  planning_result.iteration = planning_request.iteration;
203  planning_result.nominal_maneuver_valid = false;
204  planning_result.combined_maneuver_valid = false;
205  planning_result.indicator_left = false;
206  planning_result.indicator_right = false;
207 
208  three_lanes_.update();
209  ngo_.update();
210  connectionSet_.update();
211  connectionsOnLaneSource_->update();
212  connectionsOnLaneTarget_->update();
213  checkPointSet_.update();
214  checkPointsOnTargetLane_->update();
215  checkPointsOnSourceLane_->update();
216  auto current = three_lanes_.getCurrentLane();
217  auto target_lc = directionLeft_
218  ?three_lanes_.getLeftLaneChange()
219  :three_lanes_.getRightLaneChange();
220  auto target = target_lc->getTargetLane();
221 
222 
223  if(!current->isValid())
224  {
225  planning_result.status_string = "current lane invalid";
226  return;
227  }
228  if(!target->isValid())
229  {
230  planning_result.status_string = "target lane invalid";
231  return;
232  }
233  if(target_lc->getNavigationCostDifference()>pTacticalPlanner_->getMaxNavcostLoss())
234  {
235  planning_result.status_string = "lc not required for navigation";
236  // todo: investigate cost calculation as it seems implausible return;
237  }
238 
239 
240  prediction_.update();
241 
242  //delayed indicator activation
243  double s,n;
244  target->toRelativeCoordinates(planning_request.initial_state.x0ref.getX(),
245  planning_request.initial_state.x0ref.getY(),
246  s,n);
247  double sg0 = target_lc->getProgressOfGateOpen();
248  double indicator_lookahead = pTacticalPlanner_->getIndicatorLookahead();
249  if(s+indicator_lookahead>sg0)
250  {
251  planning_result.indicator_left = directionLeft_;
252  planning_result.indicator_right = !directionLeft_;
253  }
254 
255  //@TODO: Gap selection
256  // - store XY-coordinate for gap identification
257  // - remember speed of gap and time
258  // - advance XY-coordinate with speed of gap and time to get an estimate of next gap position
259  // - relocate XY-coordinate in gap center
260  // - compute s-porgress from XY-coordinate to to identify gap to basiclanechangeplanner
261  // here: naive gap selection: always use ego coordinate
262  gapX_ = planning_request.initial_state.x0ref.getX();
263  gapY_ = planning_request.initial_state.x0ref.getY();
264  double tmp;
265  target->toRelativeCoordinates(gapX_,gapY_,gaps_,tmp);
266 
267  nominal_planner_->compute(planning_request.initial_state.toMotionState());
268  if(!nominal_planner_->hasValidPlan())
269  {
270  planning_result.status_string = "nominal maneuver planning failed, "+ nominal_planner_->getStatus();
271  return;
272  }
273 
274  nominal_planner_->getSetPointRequest()->copyTo(planning_result.nominal_maneuver);
275 
276  adore::fun::RestartEffort re;
277  if(!re.isValid(planning_result.nominal_maneuver))
278  {
279  planning_result.status_string = "not restarting, maneuver too short";
280  return;
281  }
282 
283 
284  planning_result.nominal_maneuver_valid = true;
285  auto x0em = nominal_planner_->getSetPointRequest()->interpolateSetPoint(planning_request.t_emergency_start,pvehicle_);
286  bool valid_em_found = false;
287 
288  if(em_continue_active_)
289  {
290  emergency_planner2_->setJMax(-pEmergencyOperation_->getEmergencyManeuverJMin());
291  emergency_planner2_->setTStall(pEmergencyOperation_->getEmergencyManeuverTStall());
292  emergency_planner2_->setAStall(pEmergencyOperation_->getEmergencyManeuverAStall());
293  emergency_planner2_->setAMin(pEmergencyOperation_->getEmergencyManeuverAMin());
294  emergency_planner2_->compute(x0em.toMotionState());
295  if(emergency_planner2_->hasValidPlan() )
296  {
297  nominal_planner_->getSetPointRequest()->copyTo(planning_result.combined_maneuver,0);
298  planning_result.combined_maneuver.removeAfter(planning_request.t_emergency_start);
299  planning_result.combined_maneuver.setPoints.back().tEnd = planning_request.t_emergency_start;
300  emergency_planner2_->getSetPointRequest()->copyTo(planning_result.combined_maneuver,1);
301  if(collision_detection_.isValid(planning_result.combined_maneuver))
302  {
303  valid_em_found = true;
304  }
305  else
306  {
307  //planning_result.combined_maneuver.setPoints.clear();
308  }
309 
310  }
311  }
312 
313  if(em_cancel_active_)
314  {
315  if(!valid_em_found)
316  {
317  emergency_planner1_->setJMax(-pEmergencyOperation_->getEmergencyManeuverJMin());
318  emergency_planner1_->setTStall(pEmergencyOperation_->getEmergencyManeuverTStall());
319  emergency_planner1_->setAStall(pEmergencyOperation_->getEmergencyManeuverAStall());
320  emergency_planner1_->setAMin(pEmergencyOperation_->getEmergencyManeuverAMin());
321  emergency_planner1_->compute(x0em.toMotionState());
322  if(emergency_planner1_->hasValidPlan() )
323  {
324  nominal_planner_->getSetPointRequest()->copyTo(planning_result.combined_maneuver,0);
325  planning_result.combined_maneuver.removeAfter(planning_request.t_emergency_start);
326  planning_result.combined_maneuver.setPoints.back().tEnd = planning_request.t_emergency_start;
327  emergency_planner1_->getSetPointRequest()->copyTo(planning_result.combined_maneuver,1);
328  if(collision_detection_.isValid(planning_result.combined_maneuver))
329  {
330  valid_em_found = true;
331  }
332  else
333  {
334  // planning_result.combined_maneuver.setPoints.clear();
335  }
336  }
337  }
338  }
339 
340  if(!valid_em_found)
341  {
342  planning_result.status_string += "emergency maneuver planning failed";
343  return;
344  }
345 
346  double front_buffer_space = pTacticalPlanner_->getCollisionDetectionFrontBufferSpace();
347  double lateral_precision = pTacticalPlanner_->getCollisionDetectionLateralPrecision();
348  adore::fun::SetPointRequestSwath spr_swath(
349  pvehicle_->get_a()+pvehicle_->get_b()+pvehicle_->get_c()+pvehicle_->get_d()+front_buffer_space,
350  pvehicle_->get_bodyWidth(),
351  pvehicle_->get_d(),//spr reference point at rear axle
352  lateral_precision);
353  spr_swath.setLonError(pTacticalPlanner_->getCollisionDetectionLongitudinalError());
354  spr_swath.setLatError(pTacticalPlanner_->getCollisionDetectionLateralError());
355  spr_swath.append_cylinder_swath_linear(planning_result.combined_maneuver,planning_result.combined_maneuver_swath);
356  spr_swath.append_cylinder_swath_linear(planning_result.nominal_maneuver,planning_result.nominal_maneuver_swath);
357 
358  planning_result.combined_maneuver.cropAfterFirstStop(pTacticalPlanner_->getTerminateAfterFirstStopThresholdSpeed());
359  if(planning_result.combined_maneuver.setPoints.size()==0)
360  {
361  planning_result.status_string = "stopping";
362  return;
363  }
364 
365  planning_result.combined_maneuver_valid = true;
366  int laneid = directionLeft_?1:-1;
367  adore::fun::SPRNavigationCostOnLane navcostOnLane(&three_lanes_,pTacticalPlanner_,laneid);
368  planning_result.objective_values.insert({navcostOnLane.getName(),
369  navcostOnLane.getCost(planning_result.nominal_maneuver)});
370 
371  adore::fun::SPRNormalizedNavigationCost navcostNormalized(&three_lanes_,pTacticalPlanner_,laneid);
372  planning_result.objective_values.insert({navcostNormalized.getName(),
373  navcostNormalized.getCost(planning_result.nominal_maneuver)});
374 
375  adore::fun::SPRLongitudinalAcceleration2Cost lacccost;
376  planning_result.objective_values.insert({lacccost.getName(),
377  lacccost.getCost(planning_result.nominal_maneuver)});
378 
379  adore::fun::SPRLongitudinalJerk2Cost ljerkcost;
380  planning_result.objective_values.insert({ljerkcost.getName(),
381  ljerkcost.getCost(planning_result.nominal_maneuver)});
382 
383  adore::fun::SPRAverageProgressLoss progressLoss(pTacticalPlanner_);
384  planning_result.objective_values.insert({progressLoss.getName(),
385  progressLoss.getCost(planning_result.nominal_maneuver)});
386 
387  planning_result.objective_values.insert({"const_penalty",const_penalty_});
388 
389  //ttc cost
390  planning_result.objective_values.insert({ttcCost_.getName(),
391  ttcCost_.getCost(planning_result.nominal_maneuver)});
392 
393 
394  switch(pTacticalPlanner_->getCoercionPreventionStrategy())
395  {
396  case 0://turned off
397  {
398  planning_result.objective_values.insert({coercion_detection_.getName(),0.0});
399  }
400  break;
401  case 1://encode as objective value
402  {
403  planning_result.objective_values.insert({coercion_detection_.getName(),
404  coercion_detection_.isValid(planning_result.nominal_maneuver)
405  ? 0.0 : 1.0});
406  }
407  break;
408  case 2://encode as constraint
409  {
410  bool coercion_detection_passed = coercion_detection_.isValid(planning_result.nominal_maneuver);
411  planning_result.combined_maneuver_valid = coercion_detection_passed && planning_result.combined_maneuver_valid;
412  if(!coercion_detection_passed)
413  {
414  planning_result.status_string = "coercion detected for nominal maneuver";
415  }
416  }
417  break;
418  }
419 
420 
421  int number_of_time_steps = nominal_planner_->getProgressSolver().lbx().nc();
422  auto time_steps = adore::mad::linspace(nominal_planner_->getPlanningHorizon()/number_of_time_steps,nominal_planner_->getPlanningHorizon(),number_of_time_steps);
423  planning_result.nominal_maneuver_longitudinal_plan = nominal_planner_->getProgressSolver().result_fun();
424  planning_result.nominal_maneuver_longitudinal_lbx.getData() = dlib::join_cols(time_steps,nominal_planner_->getProgressSolver().lbx());
425  planning_result.nominal_maneuver_longitudinal_ubx.getData() = dlib::join_cols(time_steps,nominal_planner_->getProgressSolver().ubx());
426  planning_result.nominal_maneuver_lateral_plan = nominal_planner_->getOffsetSolver().result_fun();
427  planning_result.nominal_maneuver_lateral_lbx.getData() = dlib::join_cols(time_steps,nominal_planner_->getOffsetSolver().lbx());
428  planning_result.nominal_maneuver_lateral_ubx.getData() = dlib::join_cols(time_steps,nominal_planner_->getOffsetSolver().ubx());
429  }
430  };
431  }
